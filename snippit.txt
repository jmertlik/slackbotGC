def handle_channel_join(event):
    print("Status change for ", event['user'])
    rpgUser["user"] = event["user"]
    rpgUser["start"] = event["ts"]
    if rpgUser["start"] != None:
        rpgUser["total"] = (event["ts"] - rpgUser["start"]) + rpgUser["total"]
        rpgUser["start"] = None
